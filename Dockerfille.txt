FROM python:3.11-slim

# Instala o Chrome
RUN apt-get update && apt-get install -y \
    wget curl unzip gnupg \
    chromium chromium-driver \
    --no-install-recommends && \
    apt-get clean

WORKDIR /app
COPY requirements.txt .
RUN pip install -r requirements.txt

COPY monitor.py .

CMD ["python", "monitor.py"]
